name: Deploy to GitHub Pages

on:
  push:
    branches:
      - main  # Set the branch you want to deploy from

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest-m-nodoubtz 

    steps:
      - name: Checkout the repository
        uses: actions/checkout@v4

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: 20 # or your required version

      - name: Install dependencies
        run: npm install

      - name: Build static site
        run: npm run build # Change this if your build command is different

      - name: Deploy to GitHub Pages
        uses: peaceiris/actions-gh-pages@v4
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./build # Change this if your build output is in a different folder
